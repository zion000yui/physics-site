def diff_drive_step(x, y, theta, vr, vl, L, dt):
    v = (vr + vl) / 2
    w = (vr - vl) / L
    x += v * cos(theta) * dt
    y += v * sin(theta) * dt
    theta += w * dt
    return x, y, theta
